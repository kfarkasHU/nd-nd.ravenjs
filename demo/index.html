            </tr>
        </tbody>

const forModel = {
    objects: [
        { id: 1, text: "egy" },
        { id: 2, text: "kettő" },
        { id: 3, text: "három" },
        { id: 4, text: "négy" },
        { id: 5, text: "öt" },
        { id: 6, text: "hat" },
    ]
};

Raven.plugins.binding.fly("for-demo", forModel);
